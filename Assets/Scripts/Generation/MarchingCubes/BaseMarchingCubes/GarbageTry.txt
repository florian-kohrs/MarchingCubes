    protected static Dictionary<Tuple<AxisOffset, int>, int> mapSymmetric = new Dictionary<Tuple<AxisOffset, int>, int>();

    protected static int FindTriangulationIndexWithTriangles(IList<Vector3> tris)
    {
        int index = -1;
        for (int i = 0; i < triangulation.Count && index == -1; ++i)
        {
            List<Vector3> c = ToVector3(triangulation[i]).ToList();
            bool contained = tris.Count == c.Count;
            for (int x = 0; x < tris.Count && contained; x++)
            {
                contained = !c.TrueForAll(v => !v.SharesAnyNValuesWith(tris[x], 3));
            }
            if (contained)
            {
                index = i;
            }
        }
        return index;
    }

    public struct NeighbourKey
    {
        public NeighbourKey(int from, int tri, int to)
        {
            fromIndex = from;
            fromTriIndex = tri;
            toIndex = to;
        }

        public int fromIndex;
        public int fromTriIndex;
        public int toIndex;
    }

    private static Vector3[] ToVector3(int[] tris)
    {
        tris = tris.Where(i => i >= 0).ToArray();
        Vector3[] v = new Vector3[tris.Length / 3];
        for (int i = 0; i < tris.Length && tris[i] >= 0; i += 3)
        {
            v[i / 3] = new Vector3(tris[i], tris[i + 1], tris[i + 2]);
        }
        return v;
    }

    private static Vector3[] ToVector3(IList<int> tris)
    {
        tris = tris.Where(i => i >= 0).ToList();
        Vector3[] v = new Vector3[tris.Count / 3];
        for (int i = 0; i < tris.Count; i += 3)
        {
            v[i / 3] = new Vector3(tris[i], tris[i + 1], tris[i + 2]);
        }
        return v;
    }

    private const int SAME_VERTICES_TO_BE_NEIGHBOURS = 2;

    protected static Dictionary<NeighbourKey, List<int>> neighbourTable;

    public static Dictionary<NeighbourKey, List<int>> NeighbourTable
    {
        get
        {
            if (neighbourTable == null)
            {
                BuildNeighbourTable();
            }
            return neighbourTable;
        }
    }

    public static List<int> GetNeighbourIndicesIn(int fromIndex, int fromTriIndex, int toIndex)
    {
        return NeighbourTable[new NeighbourKey(fromIndex, fromTriIndex, toIndex)];
    }

    protected static void Add(NeighbourKey key, int i)
    {
        List<int> list;
        if (!neighbourTable.TryGetValue(key, out list))
        {
            list = new List<int>();
            neighbourTable[key] = list;
        }
        list.Add(i);
    }

    protected static void Add(NeighbourKey key, List<int> i)
    {
        List<int> list;
        if (!neighbourTable.TryGetValue(key, out list))
        {
            list = new List<int>();
            neighbourTable[key] = list;
        }
        list.AddRange(i);
    }


    protected static void BuildNeighbourTable()
    {
        neighbourTable = new Dictionary<NeighbourKey, List<int>>();

        NeighbourKey key1;
        NeighbourKey key2;

        for (int x = 0; x < triangulation.Count - 1; x++)
        {
            key1.fromIndex = x;
            key2.toIndex = x;
            for (int y = x; y < triangulation.Count; y++)
            {
                key1.toIndex = y;
                key2.fromIndex = y;
                for (int i1 = 0; i1 < triangulation[x].Length; i1 += 3)
                {
                    List<int> neighboursFrom = new List<int>();
                    key1.fromTriIndex = i1 / 3;

                    if (triangulation[x][i1] == -1)
                    {
                        break;
                    }

                    Vector3 v1 = new Vector3(
                       triangulation[x][i1],
                       triangulation[x][i1 + 1],
                       triangulation[x][i1 + 2]);

                    for (int i2 = y; i2 < triangulation[y].Length; i2 += 3)
                    {
                        Vector3 v2 = new Vector3(
                            triangulation[x][i1],
                            triangulation[x][i1 + 1],
                            triangulation[x][i1 + 2]);
                        if (i2 == -1)
                        {
                            break;
                        }

                        if (v2.SharesExactNValuesWith(v1, SAME_VERTICES_TO_BE_NEIGHBOURS))
                        {
                            key1.fromTriIndex = i1 / 3;
                            key2.fromTriIndex = i2 / 3;
                            Add(key2, key1.fromTriIndex);
                            neighboursFrom.Add(key2.fromIndex);
                        }
                    }
                    Add(key1, neighboursFrom);

                }

            }
        }
    }

    public enum AxisOffset { X, Y, Z }

    protected class RotatedTriangle
    {

        int edgeIndex1 = -1;
        int edgeIndex2 = -1;
        int edgeIndex3 = -1;

        public int edgesOnRotationSideCount;

        public bool IsFull => Count >= 3;

        public int Count
        {
            get
            {
                if (edgeIndex1 < 0)
                    return 0;
                else if (edgeIndex2 < 0)
                    return 1;
                else if (edgeIndex3 < 0)
                    return 2;
                else
                    return 3;
            }
        }


        public int this[int i]
        {
            get
            {
                switch (i)
                {
                    case 0:
                        return edgeIndex1;
                    case 1:
                        return edgeIndex2;
                    default:
                        return edgeIndex3;
                }
            }
            set
            {
                switch (i)
                {
                    case 0:
                        edgeIndex1 = value;
                        break;
                    case 1:
                        edgeIndex2 = value;
                        break;
                    default:
                        edgeIndex3 = value;
                        break;
                }
            }
        }
    }


    public static IList<int> ImportantShiftedEdges(int index, AxisOffset a)
    {
        IList<int> r = new List<int>();
        RotatedTriangle tri = new RotatedTriangle();
        for (int i = triangulation[index].Length - 1; i >= 0; i--)
        {
            int item = triangulation[index][i];

            if (item != -1)
            {
                tri.edgesOnRotationSideCount++;
                if (a == AxisOffset.X)
                {
                    if (item == 7 || item == 3)
                    {
                        tri[tri.Count] = (item - 2);
                    }
                    else if (item == 11)
                    {
                        tri[tri.Count] = (10);
                    }
                    else if (item == 8)
                    {
                        tri[tri.Count] = (9);
                    }
                    else if (item == 5 || item == 1)
                    {
                        tri[tri.Count] = (item + 2);
                    }
                    else if (item == 10)
                    {
                        tri[tri.Count] = (11);
                    }
                    else if (item == 9)
                    {
                        tri[tri.Count] = (8);
                    }
                    else
                    {
                        tri[tri.Count] = (item);
                        tri.edgesOnRotationSideCount--;
                    }
                }
                else if (a == AxisOffset.Y)
                {
                    if (item < 4 && item >= 0)
                    {
                        tri[tri.Count] = (item + 4);
                    }
                    else if (item >= 4 && item < 8)
                    {
                        tri[tri.Count] = (item - 4);
                    }
                    else
                    {
                        tri[tri.Count] = (item);
                        tri.edgesOnRotationSideCount--;
                    }
                }
                else if (a == AxisOffset.Z)
                {
                    if (item == 2 || item == 6)
                    {
                        tri[tri.Count] = (item - 2);
                    }
                    else if (item == 11)
                    {
                        tri[tri.Count] = (8);
                    }
                    else if (item == 10)
                    {
                        tri[tri.Count] = (9);
                    }
                    else if (item == 4 || item == 0)
                    {
                        tri[tri.Count] = (item + 2);
                    }
                    else if (item == 8)
                    {
                        tri[tri.Count] = (11);
                    }
                    else if (item == 9)
                    {
                        tri[tri.Count] = (10);
                    }
                    else
                    {
                        tri[tri.Count] = (item);
                        tri.edgesOnRotationSideCount--;
                    }
                }
            }
            if (tri.IsFull)
            {
                if(tri.edgesOnRotationSideCount == 3)
                {
                    for (int x = 0; x < tri.Count; x++)
                    {
                        r.Add(tri[x]);
                    }
                }
                else
                {
                    for (int x = tri.Count - 1; x >= 0; x--)
                    {
                        r.Add(tri[x]);
                    }
                }
                tri = new RotatedTriangle();
            }
        }
        return r;
    }
